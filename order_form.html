  <script>

    // ====== CONFIG (keep everything else static; change these only if you want to) ======

    const CONFIG = {

      availableToTradeUsd: 10000.0,   // displayed only

      currentLongShares: 100.0,       // user currently LONG

      pricePerShareUsd: 0.50,         // 50 cents

      contractPayoutUsd: 1.00,        // used for "To win"

      sliderMaxShares: 500

    };


    // ====== HELPERS ======

    const $ = (id) => document.getElementById(id);


    function fmtUsd(x){

      // keep 1 decimal like the screenshot ($25.0)

      const v = Math.round(x * 10) / 10;

      return "$" + v.toLocaleString(undefined, { minimumFractionDigits: 1, maximumFractionDigits: 1 });

    }


    function fmtNum1(x){

      const v = Math.round(x * 10) / 10;

      return v.toLocaleString(undefined, { minimumFractionDigits: 1, maximumFractionDigits: 1 });

    }


    function clamp(n, min, max){ return Math.max(min, Math.min(max, n)); }


    // ====== CORE LOGIC (matches your described behavior) ======

    // Selling quantity Q against an existing long L:

    // - Closes min(Q, L) long shares

    // - If Q > L, opens (Q - L) new short shares

    // UI:

    // - To receive = (Q - L)+ * price

    // - To pay = 0 (shown as --)

    // - Net Position = L - Q (positive => LONG, negative => SHORT)

    // - To win = abs(net) * $1

    function recompute(){

      const L = CONFIG.currentLongShares;

      const price = CONFIG.pricePerShareUsd;

      const payout = CONFIG.contractPayoutUsd;


      const Qraw = parseFloat($("sharesInput").value || "0");

      const Q = clamp(isFinite(Qraw) ? Qraw : 0, 0, CONFIG.sliderMaxShares);


      // keep slider + input in sync

      if (parseFloat($("sharesSlider").value) !== Q) $("sharesSlider").value = Q;

      if (parseFloat($("sharesInput").value) !== Q) $("sharesInput").value = Q;


      // net after sell

      const net = L - Q;                 // + => long, - => short

      const netAbs = Math.abs(net);


      // extra short opened (only beyond closing existing long)

      const extraShort = Math.max(Q - L, 0);


      const toReceive = extraShort * price;

      const toWin = netAbs * payout;


      // Render top stats

      $("available").textContent = fmtUsd(CONFIG.availableToTradeUsd);

      $("currentShares").textContent = fmtNum1(L);

      $("avgPrice").textContent = (price * 100).toFixed(1) + "¢";

      $("matching").textContent = fmtNum1(Q);


      // Render pay/receive

      $("toPay").textContent = "--";

      $("toReceive").textContent = fmtUsd(toReceive);


      // Render net position block

      $("netShares").textContent = fmtNum1(netAbs);

      $("toWin").textContent = fmtUsd(toWin);


      const netSideEl = $("netSide");

      if (net < 0){

        netSideEl.textContent = "SHORT";

        netSideEl.classList.remove("long");

        netSideEl.classList.add("short");

      } else if (net > 0){

        netSideEl.textContent = "LONG";

        netSideEl.classList.remove("short");

        netSideEl.classList.add("long");

      } else {

        // flat: show SHORT style off; keep label as FLAT

        netSideEl.textContent = "FLAT";

        netSideEl.classList.remove("short");

        netSideEl.classList.remove("long");

      }


      // Percentage box (purely cosmetic; keep it stable-ish like screenshot)

      // Here: show Q as % of slider max.

      const pct = Math.round((Q / CONFIG.sliderMaxShares) * 100);

      $("pctBox").textContent = pct + "%";


      // Reduce-only behavior (optional UX): if enabled, cap Q at current long (cannot go net short)

      if ($("reduceOnly").checked && Q > L){

        $("sharesInput").value = L;

        $("sharesSlider").value = L;

        // recompute once after enforcing cap

        $("reduceOnly").dataset.enforced = "1";

        return recompute();

      }

    }


    // ====== INIT ======

    $("sharesSlider").max = String(CONFIG.sliderMaxShares);

    $("sharesSlider").value = $("sharesInput").value;


    $("sharesInput").addEventListener("input", recompute);

    $("sharesSlider").addEventListener("input", () => {

      $("sharesInput").value = $("sharesSlider").value;

      recompute();

    });

    $("reduceOnly").addEventListener("change", recompute);


    recompute();

  </script>
